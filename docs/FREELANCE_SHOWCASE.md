Create a simple Gradio or Streamlit app:

```python
# demo_app.py
import gradio as gr
from src.inference import load_base_model, summarize_dialogue

model, tokenizer = load_base_model()

def summarize(dialogue):
    return summarize_dialogue(model, tokenizer, dialogue)

# Professional examples
examples = [
    ["Customer: My order #12345 hasn't arrived. It's been 2 weeks.\nAgent: I apologize for the delay. Let me check... The package shows as delivered to your address on May 15th.\nCustomer: I never received it.\nAgent: I'll file a claim and send a replacement immediately."],
    ["Sales Rep: Thanks for your time today. Can you tell me about your current CRM?\nClient: We use Salesforce but struggle with custom reports.\nSales Rep: Our tool integrates directly with Salesforce and provides 50+ pre-built reports.\nClient: That sounds interesting. What's the pricing?"],
]

interface = gr.Interface(
    fn=summarize,
    inputs=gr.Textbox(lines=8, label="Paste Your Conversation"),
    outputs=gr.Textbox(label="AI-Generated Summary"),
    examples=examples,
    title="AI Conversation Summarizer - Free Demo",
    description="See how AI can save your team hours every day. Paste any conversation below.",
    article="<a href='mailto:you@email.com'>Contact me for custom solutions for your business</a>"
)

interface.launch()
```
